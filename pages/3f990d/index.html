# 7.3 返回Map

  • 当返回的值没有合适的pojo类来接收时，可以使用Map集合
  Map<String,Object> selectByIdRetMap(Long id);
1
    <select id="selectByIdRetMap" resultType="map">
        select * from t_car where id = #{id}
    </select>

    @Test
    public void testSelectByIdRetMap(){
        SqlSession sqlSession = SqlSessionUtil.openSession();
        CarMapper mapper = sqlSession.getMapper(CarMapper.class);

        Map<String, Object> stringObjectMap = mapper.selectByIdRetMap(165L);

        System.out.println(stringObjectMap);
        // {car_num=6666, id=165, guide_price=32.00, produce_time=2020-11-11T00:00, brand=丰田霸道, car_type=燃油车}

# 7.5 resultMap结果映射

  • 查询查来的结果的列名与Java对象的属性名对应不上

    • 第一种方案：as给列名取别名
    • 第二种方案：使用resultMap进行结果映射
    • 第三种方案：是否开启驼峰式命名自动映射

  • 使用resultMap结果映射

     List<Car> findAll();
    1
        <!-- 指定结果集映射，数据库表的列名与属性名的对应关系 -->
    <!-- type属性：用来指定POJO类的类名
         id属性：指定resultMap的唯一标识，将来这个要在select标签中使用-->
    <resultMap id="carResultMap" type="car">
        <!--如果数据库表中有主键，可配置一个id标签 -->
        <id property="id" column="id"></id>
        <!--property后面填写POJO类的属性名 -->
        <!--column后面填写数据库表的字段名 -->
        <result property="carNum" column="car_num" javaType="string" jdbcType="VARCHAR"/>
        <!-- 如果数据库表的列名与属性名一致，可以省略不写-->
        <!-- <result property="brand" column="brand"/>-->
        <result property="guidePrice" column="guide_price"/>
        <result property="produceTime" column="produce_time"/>
        <result property="carType" column="car_type"/>
    </resultMap>

    <!--select标签的resultMap属性，用来指定使用哪个结果集映射，resultMap后面的值是resultMap的id-->
    <select id="findAll" resultMap="carResultMap">
        select * from t_car;
    </select>
