  • 业务实现
/**
 * @Author: yli
 * @Date: 2023/11/20/10:29
 * @Description:
 */
@Component
// 开启jms注解
@EnableJms
public class QueueProduce {

    @Autowired
    private JmsMessagingTemplate jmsMessagingTemplate;

    @Autowired
    private Queue queue;

    // 点击触发投送
    public void produceMessages(){
        jmsMessagingTemplate.convertAndSend(queue, UUID.randomUUID().toString().
                replaceAll("-","").substring(0,6));
        System.out.println("*********over*********");
    }

    // 间隔时间3秒投递
    @Scheduled(fixedDelay = 3000)
    public void produceMsgScheduled(){
        jmsMessagingTemplate.convertAndSend(queue, "Scheduled：" + UUID.randomUUID().toString().
                replaceAll("-","").substring(0,6));
        System.out.println("*********Scheduled over*********");
    }
}
