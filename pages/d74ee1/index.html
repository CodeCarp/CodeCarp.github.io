# 6、MyBatis参数处理

# 6.1 单个简单类型参数

  • 简单类型包括：
    • byte short int long float double char
    • Byte Short Integer Long Float Double Character
    • String
    • java.util.Date
    • java.sql.Date

# 6.1.1 单个参数之参数为Long类型

<!-- parameterType属性的作用：
        参数的类型，mybatis框架自身带有自动推断机制，大部分情况下可以省略不写-->
    <!-- List<Student> selectById(Long id);   -->
    <select id="selectById" resultType="student" parameterType="long">
        select * from t_student where id = #{id}
    </select>

    public void testSelectById(){
        SqlSession sqlSession = SqlSessionUtil.openSession();
        StudentMapper mapper = sqlSession.getMapper(StudentMapper.class);
        List<Student> students = mapper.selectById(1L);
        students.forEach(student -> System.out.println(student));
        sqlSession.close();

    @Test
    public void testSelectByBirth() throws Exception{
        SqlSession sqlSession = SqlSessionUtil.openSession();
        StudentMapper mapper = sqlSession.getMapper(StudentMapper.class);

        SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
        Date birth = sdf.parse("1980-10-11");

        List<Student> students = mapper.selectByBirth(birth);
        students.forEach(student -> System.out.println(student));
        sqlSession.close();
    }

# 6.2 参数是Map集合

<insert id="insertStudentByMap" parameterType="map">
   insert into t_student(id,name,age,sex,birth,height) values(null,#{name},#{age},#{sex},#{birth},#{height})
</insert>
1
2
3
	@Test
    public void testInsertStudentByMap() throws Exception{
        SqlSession sqlSession = SqlSessionUtil.openSession();
        StudentMapper mapper = sqlSession.getMapper(StudentMapper.class);

        Map<String,Object> map = new HashMap<>();
        map.put("name", "张无忌");
        map.put("age", "20");
        map.put("height", "1.78");
        map.put("sex", '男');
        map.put("birth", new Date());

        mapper.insertStudentByMap(map);
        sqlSession.commit();
        sqlSession.close();
    }

# 6.4 多个参数的处理

  • 如果是多个参数的话，mybatis底层是如何做的呢？

    • mybatis框架会自动创建一个Map集合，并且Map集合是以这种方式存储参数的

      map.put("arg0",name);
map.put("arg1",sex);
map.put("arg..",...);

map.put("param1",name);
map.put("param2",sex);
map.put("param..",...);
