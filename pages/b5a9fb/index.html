# 第四章 Vue中的Ajax配置代理slot插槽

# 4.1 Vue脚手架配置代理

需下载axios库 npm install axios

vue.config.js是一个可选的配置文件，如果项目的（和package.json同级的）根目录中存在这个文件，那么他会被@vue/cli-servive自动加载，也可以使用package.json中的vue字段，但是注意这种写法需要严格遵照JSON的格式来写

方式一

在vue.config.js中添加如下配置

module.exports = {
    devServer:{
        proxy:'http://localhost:需要代理的端口号'
    }
}

说明：

  • 优点：配置简单，请求资源时直接发给前端（8080）即可
  • 缺点：不能配置多个代理，不能灵活的控制请求是否走代理
  • 工作方式：若按照上述配置代理，当请求了前端不存在的资源时，才会将请求转发给服务器（优先匹配前端资源）

		name:'App',
		methods:{
			getStudents(){
				axios.get('http://localhost:8080/students').then(
					response =>{
						console.log('请求成功，',response.data)
					},
					error =>{
						console.log('请求失败',error.message)
					}
				)
			}
		}
	}
</script>

方式二

module.exports = {
	devServer: {
      proxy: {
      '/api1': {// 匹配所有以 '/api1'开头的请求路径
        target: 'http://localhost:5000',// 代理目标的基础路径
        changeOrigin: true,
        pathRewrite: {'^/api1': ''}
      },
      '/api2': {// 匹配所有以 '/api2'开头的请求路径
        target: 'http://localhost:5001',// 代理目标的基础路径
        changeOrigin: true,
        pathRewrite: {'^/api2': ''}
      }
    }
  }
}
/*
   changeOrigin设置为true时，服务器收到的请求头中的host为：localhost:5000
   changeOrigin设置为false时，服务器收到的请求头中的host为：localhost:8080
   changeOrigin默认值为true
*/

vue.config.js

const { defineConfig } = require('@vue/cli-service')
module.exports = defineConfig({
  transpileDependencies: true,
  lintOnSave:false,  // 关闭语法检测
  // 开启代理服务器（方式一）
  /* devServer:{
    proxy:'http://localhost:5000'
  } */
  devServer:{
    proxy:{
      '/yli':{
        target:'http://localhost:5000',
        pathRewrite:{'^/yli':''},
        ws:true, // 用于支持websocket
        changerOrigin:true // 用于控制请求头中的host值
      },
      '/demo':{
        target:'http://localhost:5001',
        pathRewrite:{'^/demo':''},
        ws:true, // 用于支持websocket
        changerOrigin:true // 用于控制请求头中的host值
      }
    }
  }
})

src/components/Search.vue

<template>
    <section class="jumbotron">
      <h3 class="jumbotron-heading">Search Github Users</h3>
      <div>
        <input type="text" placeholder="enter the name you search" v-model="KeyWord"/>&nbsp;
        <button @click="searchUsers">Search</button>
      </div>
    </section>
</template>

<script>
    import axios from 'axios'
    export default {
        name:'Search',
        data(){
            return {
                KeyWord:''
            }
        },
        methods:{
            searchUsers(){
                //  请求前更新List数据
                this.$bus.$emit('updateListData',{isFirst:false,isLoading:true,errMsg:'',users:[]})
                axios.get(`https://api.github.com/search/users?q=${this.KeyWord}`).then(
                    response =>{
                        // console.log('请求成功',response.data.items)
                        // this.$bus.$emit('getUsers',response.data.items)
                        this.$bus.$emit('updateListData',{isLoading:false,errMsg:'',users:response.data.items})

                    },
                    error => {
                        // console.log('请求失败，',error.message)
                        this.$bus.$emit('updateListData',{isLoading:false,errMsg:error.message,users:[]})
                    }
                )
            }
        }
    }
</script>

            })
        }
    }
</script>

# 4.3 vue-resource

Vue项目常用两个Ajax库

  1. axios：通常的ajax请求库，官方推荐，效率高
  2. vue-resource：vue插件库，vue1.x使用广泛，官方已经不再维护

下载vue-resource库 npm i vue-resource

# 4.4 slot插槽

<slot>插槽，让父组件可以向子组件指定位置插入html结构，也是一种组件间通信的方式

适用于：父组件===》子组件

  1. 分类：默认插槽、具名插槽、作用域插槽
  2. 使用方式：
    • 默认插槽
父组件中：
<Category>
	<div>
        html结构
    </div>
</Category>
子组件中：
<template>
	<div>
        <!--定义插槽-->
        <slot>插槽默认内容</slot>
    </div>
</template>

作用域插槽

scope用于父组件往子组件插槽放的html结构接收子组件的数据

理解：数据在组件的自身，但根据数据生成的结构需要组件的使用者来决定

父组件中：
		<Category>
			<template scope="scopeData">
				<!-- 生成的是ul列表 -->
				<ul>
					<li v-for="g in scopeData.games" :key="g">{{g}}</li>
				</ul>
			</template>
		</Category>

		<Category>
			<template slot-scope="scopeData">
				<!-- 生成的是h4标题 -->
				<h4 v-for="g in scopeData.games" :key="g">{{g}}</h4>
			</template>
		</Category>
子组件中：
        <template>
            <div>
                <slot :games="games"></slot>
            </div>
        </template>
		
        <script>
            export default {
                name:'Category',
                props:['title'],
                //数据在子组件自身
                data() {
                    return {
                        games:['红色警戒','穿越火线','劲舞团','超级玛丽']
                    }
                },
            }
        </script>
