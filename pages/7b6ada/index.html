<template>
  <!-- 导入-->
  <el-dialog :visible.sync="uploadDialog" center :before-close="importBeforeClose">
    <el-upload
        ref="upload"
        action
        :accept="acceptFileType"
        :limit="1"
        :multiple="false"
        :file-list="excelFiles"
        :on-change="handleChangeFile"
        :auto-upload="false"
        style="text-align: center; margin-bottom: 20px">
      <i class="el-icon-upload"></i>
      <div class="el-upload__text">
        <em>点击上传</em>
      </div>
    </el-upload>
    <span slot="footer" class="dialog-footer">
                <el-button
                    @click="submitUpload"
                    type="primary"
                    size="mini"
                    :loading="uploadLoading">确定导入
                </el-button>
                <el-button @click="importBeforeClose" size="mini">取消</el-button>
            </span>
    <div v-if="respMessageData.length > 0">
      <a href="javascript:void(0)" @click="downloadFailImportData">点击下载导入失败的excel数据</a>
      <div class="infinite-list-wrapper">
        <ul
            class="list"
            v-infinite-scroll="load"
            :infinite-scroll-disabled="disabled"
            :infinite-scroll-immediate="false">
          <li v-for="(item,index) in showRespMsg" :key="index" class="list-item">{{ item }}</li>
        </ul>
        <p v-if="loading">加载中...</p>
        <p v-if="noMore">没有更多了</p>
      </div>
    </div>

  </el-dialog>
</template>

<script>
import axios from "@/axios/index";
import {exportExcel} from '@/assets/js/ArrayDataExport.js'
export default {
  name: 'upload',
  props:{
    uploadDialog:{
      type: Boolean,
      default: false,
    },
    requestUrl:{
      type:String,
      default: ""
    },
    fieldKeys:{
      type: Array,
      default:[],
    },
    fieldLabels:{
      type: Array,
      default:[]
    },
    fileName:{
      type:String,
      default: ""
    },
    queryData:{
      type: Function,
      default:()=>{}
    }
  },
  data() {
    return {
      // 控制对话框显示
      uploadDialog: false,
      // 导入文件类型
      acceptFileType: ".xls,.xlsx",
      // 导入文件
      excelFiles: [],
      // 上传文件
      excelFile: {},
      // 加载
      uploadLoading: false,
      // 接收返回的消息（导入数据失败的信息）
      respMessageData: [],
      loading:false,
      showRespMsg:[],
      count:8,
      resultData:[],
    }
  },
  computed:{
    noMore(){
      return this.respMessageData.length == this.showRespMsg.length;
    },
    disabled(){
      return this.loading || this.noMore;
    }
  },
  methods: {
    load(){
      this.loading = true;
      setTimeout(()=>{
        this.count += 3;
        this.showRespMsg = this.respMessageData.slice(0,this.count);
        this.loading = false;
      },1000)
    },
    // 导入弹窗关闭回调
    importBeforeClose() {
      this.$emit("update:uploadDialog", false);
      this.excelFiles = [];
      if (this.respMessageData.length > 0) {
        this.queryData();
        this.respMessageData = [];
      }
    },
    // 添加文件回调
    handleChangeFile(file, fileList) {
      this.excelFile = file;
      this.excelFiles = [];
      this.excelFiles.push(file);
    },
    // 导入显示对话框
    handleImport() {
      this.uploadDialog = true;
    },
    // 点击下载导入失败的数据
    downloadFailImportData(){
      exportExcel(this.resultData[0], this.fieldLabels, this.fieldKeys, this.fileName);
    },
    // 确认导入按钮
    submitUpload() {
      this.uploadLoading = true;
      if (this.excelFiles && this.excelFiles.length > 0) {
        let formData = new FormData();
        formData.append("file", this.excelFile.raw);
        axios({
          method: "post",
          headers: {
            'Content-Type': 'multipart/form-data',
          },
          url: this.requestUrl,
          data: formData
        }).then(res => {
          this.uploadLoading = false;
          let resultMsg = res.data.data.msg.slice(0, res.data.data.msg.length - 1);
          this.resultData = res.data.data.data;
          if (this.resultData.length > 1) {
            this.$message.success("数据导入成功，导入失败数据已返回");
          } else {
            this.$message.success("数据全部导入成功")
            this.uploadDialog = false;
            this.queryData();
          }
          this.respMessageData = [];
          this.respMessageData = resultMsg;
          this.showRespMsg = [];
          this.showRespMsg = resultMsg.slice(0,8);
        }).catch(err => {
          this.uploadLoading = false;
          this.$message.error('导入异常:' + err);
        });
      }
      // excelFiles为空，没有选择文件
      else {
        this.uploadLoading = false;
        this.$message.error('请选择文件');
      }
    }
  }
}

</script>

# 下载导入失败数据的excel的方法

/**
 * 导出表格数据到 Excel 文件
 * @param {Array} tableData - 表格数据
 * @param {Array} fieldLabels - 表头组成的数组
 * @param {Array} fieldKeys - 列属性名组成的数组
 * @param {String} fileName - 导出的文件名
 */
function exportExcel(tableData, fieldLabels, fieldKeys, fileName) {
    let dataStr = fieldLabels.toString() + '\r\n';
    tableData.forEach(item => {
        fieldKeys.forEach(key => {
            console.log(item[key])
            // 加引号是为了使换行符在单元格内正常显示
            dataStr += `"${item[key]}"\t,`;
        });
        dataStr += '\r\n';
    });

    // encodeURIComponent 解决中文乱码
    const url = "data:text/xls;charset=utf-8,\ufeff" + encodeURIComponent(dataStr);
    const link = document.createElement("a");
    link.href = url;
    link.download = fileName + ".xls";
    link.style.display = 'none';
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link); //释放标签
}

export {
    exportExcel
}

# 后端导入监听器

package cn.com.victorysoft.business.specialEquip.listenter;

import cn.com.victorysoft.business.specialEquip.entity.ImportPressureVessel;
import cn.com.victorysoft.business.specialEquip.exception.CustomExcelAnalysisException;
import cn.com.victorysoft.vseaf.core.repository.MybatisRepository;
import com.alibaba.excel.context.AnalysisContext;
import com.alibaba.excel.event.AnalysisEventListener;
import com.alibaba.excel.exception.ExcelAnalysisException;
import com.alibaba.excel.exception.ExcelDataConvertException;

import javax.validation.ConstraintViolation;
import javax.validation.Validation;
import javax.validation.Validator;
import java.lang.reflect.Method;
import java.text.MessageFormat;
import java.util.*;

/**
 * @Author: yli
 * @CreateTime: 2023-09-06  11:11
 * @Description: 处理导入的excel数据
 */
public class  ImportListener<T> extends AnalysisEventListener<T> {

    private String NAME_SPACE;


    private MybatisRepository mybatisRepository;

    /**
     * 读取五百条数据，批量插入一次。
     */
    private static final int BATCH_COUNT = 500;

    /**
     * 记录Excel读取的行数(成功)
     */
    private int count = 0;

    /**
     * 数据读取进度
     */
    private int pointer = 0;

    /**
     * 保存读取成功的Excel数据
     */
    private List<T> entities = new ArrayList<>(BATCH_COUNT * 2);

    private List<T> failData;

    /**
     * 保存读取失败的Excel消息
     */
    private List<String> errorMessage = new ArrayList<>();

    /**
     * 行头
     */
    private Map headMap = new HashMap<>();

    /**
     * Springboot参数校验
     */
    private static Validator validator = Validation.buildDefaultValidatorFactory().getValidator();

    /**
     * 构造函数初始化监听器
     */
    public ImportListener(String NAME_SPACE, MybatisRepository mybatisRepository) {
        this.NAME_SPACE = NAME_SPACE;
        this.mybatisRepository = mybatisRepository;
        this.failData = new ArrayList<>();
    }


    /**
     * 获取表头名
     *
     * @param headMap
     * @param context
     */
    @Override
    public void invokeHeadMap(Map headMap, AnalysisContext context) {
        this.headMap = headMap;
    }

    //利用SpringBoot自带的注解进行参数校验
    public void validate(T importData) {
        Set<ConstraintViolation<T>> validate = validator.validate(importData);
        if (validate.size() > 0) { // 校验不通过的情况
            StringBuilder sb = new StringBuilder();
            validate.forEach(it -> {
                sb.append(it.getMessage());
                sb.append("\t");
            });
            failData.add(importData);
            this.pointer += 1;
            throw new ExcelAnalysisException(sb.toString());
        } else {
            //importPressureVessel.setId(UUID.randomUUID().toString().replaceAll("-", ""));
            try {
                Method setId = importData.getClass().getMethod("setId", String.class);
                setId.invoke(importData, UUID.randomUUID().toString().replaceAll("-", ""));
            }catch (Exception e){
                errorMessage.add("类型异常");
            }
            this.count = this.count + 1;
            this.pointer += 1;
        }
    }

    /**
     * 解析到一条数据,对每条数据进行操作
     */
    @Override
    public void invoke(T importData, AnalysisContext analysisContext) {
        validate(importData);
        entities.add(importData);
        if (entities != null && entities.size() >= BATCH_COUNT) {
            saveData();
            entities.clear();
        }
    }

    /**
     * 在转换异常 获取其他异常下会调用本接口。抛出异常则停止读取。如果这里不抛出异常则 继续读取下一行。
     *
     * @param exception
     * @param context
     * @throws Exception
     */
    @Override
    public void onException(Exception exception, AnalysisContext context) {
        String mes = null;
        if (exception instanceof ExcelDataConvertException) {
            //转换格式发生的异常,不会进入invoke解析,这里要++ 把解析失败数据也加上
            this.count = this.count + 1;
            this.pointer += 1;
            ExcelDataConvertException excelDataConvertException = (ExcelDataConvertException) exception;
            // 获取列名
            String columnName = (String) this.headMap.get(excelDataConvertException.getColumnIndex());
            // 获取行
            int row = excelDataConvertException.getRowIndex();
            // 获取出错的单元块
            String cellData = excelDataConvertException.getCellData().toString();
            mes = MessageFormat.format("错误发生在{0}行,{1}列,数据为:{2}", (row + 1), columnName, cellData);
        }
        // 自定义抛出的异常
        else {
            mes = MessageFormat.format("错误发生在{0}行,原因为:{1},", this.pointer + 1, exception.getMessage());
        }
        errorMessage.add(mes);
    }

    /**
     * 读取所有数据完毕进行的操作
     */
    @Override
    public void doAfterAllAnalysed(AnalysisContext analysisContext) {
        if (entities.size() > 0) {
            saveData();
            entities.clear();
        }
        StringBuilder sb = new StringBuilder();
        String countMessage = MessageFormat.format(
                "一共{0}条数据，成功导入{1}条数据，重复数据不会插入:",
                this.pointer, this.count);
        sb.append(countMessage);
        //将错误信息用字符串返回
        if (this.errorMessage.size() > 0) {
            errorMessage.forEach((it) -> {
                sb.append("|");
                sb.append(it);
            });
        }
        throw new CustomExcelAnalysisException(sb.toString(),failData);
    }

    /**
     * 具体执行读取到的excel数据进行存储的逻辑，比如写入到mysql等操作
     */
    private void saveData() {
        mybatisRepository.insert(this.NAME_SPACE, entities);
    }
}

# 自定义异常捕捉

/**
 * @Author: yli
 * @CreateTime: 2023-09-06  17:32
 * @Description: 自定义异常类，用于抛出异常信息和异常数据
 */
public class CustomExcelAnalysisException extends ExcelAnalysisException {
    private List failData;

    public CustomExcelAnalysisException(String message, List failData) {
        super(message);
        this.failData = failData;
    }

    public List getFailData(){
        return failData;
    }
}
